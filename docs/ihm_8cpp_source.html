523 {

525 
526  if(preferences.size() > 0)
527  {
528  ui->selectionLongueurPreparation->setValue(
529  preferences

532  .toInt() -
533  1);
534  gererLongueurPreparation(preferences
535  .at(Cafetiere::ChampsTablePreferences::
536  COLONNE_PREFERENCES_ID_TYPEBOISSON)
537  .toInt() -
538  1);
539  switch(
540  preferences
541  .at(
542  Cafetiere::ChampsTablePreferences::COLONNE_PREFERENCES_ID_CAPSULE)
543  .toInt() -
544  1)
545  {
546  case Cafetiere::NomCaspule::Colombia:
547  selectionnerColombia();
548  break;
549  case Cafetiere::NomCaspule::Indonesia:
